Introducción a Quark ORM

Quark ORM es un micro-ORM de alto rendimiento para Go que ofrece una API type-safe y minimalista. En este showcase, utilizamos Quark para todas las operaciones de base de datos.

🚀 Características Principales

  • Type-Safe: API fuertemente tipada con generics
  • Alto Rendimiento: 95% del rendimiento de SQL raw
  • Minimalista: Sin magia, código predecible
  • Multi-Dialect: SQLite, PostgreSQL, MySQL
  • Migraciones: Auto-migración de modelos

Inicialización del Cliente

import (
    "github.com/jcsvwinston/quark"
    _ "modernc.org/sqlite"
)

// Crear cliente Quark
db, err := sql.Open("sqlite", "app.db")
client, err := quark.New(db, quark.WithDialect(quark.SQLite()))

// Migrar modelos
err = client.Migrate(ctx,
    &models.Article{},
    &models.Author{},
    &models.Category{},
)

Consultas Básicas

List - Obtener múltiples registros

articles, err := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    OrderBy("published_at", "DESC").
    Limit(10).
    List()

First - Obtener primer registro

article, err := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    OrderBy("published_at", "DESC").
    First()

Count - Contar registros

count, err := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    Count()

Where - Filtros básicos

// Igualdad
.Where("published", "=", true)

// Mayor que
.Where("view_count", ">", 100)

// Menor que
.Where("view_count", "<", 1000)

// Diferente de
.Where("status", "!=", "deleted")

// Filtros encadenados
.Where("published", "=", true).
Where("view_count", ">", 500)

OrderBy - Ordenamiento

// Ascendente
.OrderBy("published_at", "ASC")

// Descendente
.OrderBy("published_at", "DESC")

// Múltiples campos (simulado con Where encadenado)

Limit - Limitar resultados

articles, err := quark.For[models.Article](ctx, client).
    Limit(10).
    List()

Consultas Avanzadas

Like - Búsqueda de texto parcial

articles, err := quark.For[models.Article](ctx, client).
    Where("title", "LIKE", "%Go%").
    List()

// Contiene
.Where("title", "LIKE", "%Go%")

// Empieza con
.Where("title", "LIKE", "Go%")

// Termina con
.Where("title", "LIKE", "%Go")

In - Lista de valores

articles, err := quark.For[models.Article](ctx, client).
    Where("category_id", "IN", []int64{1, 2, 3}).
    List()

// NotIn
.Where("category_id", "NOT IN", []int64{4, 5})

Between - Rango de valores

articles, err := quark.For[models.Article](ctx, client).
    Where("published_at", ">=", startDate).
    Where("published_at", "<=", endDate).
    List()

// Rango numérico
.Where("view_count", ">=", 100).
Where("view_count", "<=", 1000)

IsNull / IsNotNull - Valores nulos

// Es nulo
.Where("deleted_at", "IS NULL")

// No es nulo
.Where("deleted_at", "IS NOT NULL")

Agregaciones

Sum - Suma de valores

// Calcular suma manualmente
articles, _ := quark.For[models.Article](ctx, client).List()
totalViews := 0
for _, art := range articles {
    totalViews += art.ViewCount
}

Avg - Promedio

// Calcular promedio manualmente
articles, _ := quark.For[models.Article](ctx, client).List()
totalViews := 0
for _, art := range articles {
    totalViews += art.ViewCount
}
avgViews := totalViews / len(articles)

Min / Max - Valores extremos

articles, _ := quark.For[models.Article](ctx, client).List()

// Min
minViews := articles[0].ViewCount
for _, art := range articles {
    if art.ViewCount < minViews {
        minViews = art.ViewCount
    }
}

// Max
maxViews := articles[0].ViewCount
for _, art := range articles {
    if art.ViewCount > maxViews {
        maxViews = art.ViewCount
    }
}

Modificaciones

Create - Crear registro

article := models.Article{
    Title:       "Nuevo Artículo",
    Slug:        "nuevo-articulo",
    Summary:     "Resumen del artículo",
    Content:     "Contenido completo",
    Published:   true,
    PublishedAt: time.Now(),
    ViewCount:   0,
    AuthorID:    1,
    CategoryID:  1,
    CreatedAt:   time.Now(),
    UpdatedAt:   time.Now(),
}

err := quark.For[models.Article](ctx, client).Create(&article)

Update - Actualizar registro

// Primero obtener el registro
articles, err := quark.For[models.Article](ctx, client).
    Where("id", "=", id).
    List()

if err == nil && len(articles) > 0 {
    article := articles[0]
    article.ViewCount++
    article.UpdatedAt = time.Now()
    
    _, err = quark.For[models.Article](ctx, client).Update(&article)
}

Delete - Eliminar registro

// Primero obtener el registro
articles, err := quark.For[models.Article](ctx, client).
    Where("id", "=", id).
    List()

if err == nil && len(articles) > 0 {
    _, err = quark.For[models.Article](ctx, client).Delete(&articles[0])
}

Relaciones

Quark ORM permite trabajar con relaciones mediante consultas separadas. En este showcase, implementamos relaciones manuales entre modelos.

Relación Article → Author

// Obtener artículo
articles, err := quark.For[models.Article](ctx, client).
    Where("id", "=", articleID).
    List()

// Obtener author relacionado
authors, err := quark.For[models.Author](ctx, client).
    Where("id", "=", articles[0].AuthorID).
    List()

if len(authors) > 0 {
    author := authors[0]
    // Usar author...
}

Relación Article → Category

categories, err := quark.For[models.Category](ctx, client).
    Where("id", "=", articles[0].CategoryID).
    List()

Relación Article → Comments (One-to-Many)

comments, err := quark.For[models.Comment](ctx, client).
    Where("article_id", "=", articles[0].ID).
    Where("approved", "=", true).
    List()

Paginación

Paginación con Limit

page := 1
pageSize := 10
offset := (page - 1) * pageSize

// Obtener total de registros
total, _ := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    Count()

// Obtener registros paginados
articles, err := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    OrderBy("published_at", "DESC").
    Limit(pageSize).
    List()

// Calcular metadatos de paginación
totalPages := (int(total) + pageSize - 1) / pageSize
hasNext := page < totalPages
hasPrev := page > 1

Transacciones

Las transacciones permiten agrupar múltiples operaciones en una unidad atómica. Si alguna operación falla, todas se revierten (rollback).

Transacción Simple

// Crear artículo y comentario en transacción
// Nota: Implementación simplificada - Quark puede tener API específica
article := models.Article{...}
comment := models.Comment{...}

// Crear artículo
if err := quark.For[models.Article](ctx, client).Create(&article); err != nil {
    return err // Rollback si falla
}

// Crear comentario
if err := quark.For[models.Comment](ctx, client).Create(&comment); err != nil {
    // En una transacción real, haríamos rollback aquí
    return err
}

// Commit exitoso
