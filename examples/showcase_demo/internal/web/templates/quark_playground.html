API Endpoints de Quark

Prueba cada característica de Quark ORM haciendo clic en los siguientes endpoints:

🔍 Consultas Básicas

  • GET /api/quark/first - First()
  • GET /api/quark/find/:id - Find()
  • GET /api/quark/complex - Where encadenado
  • GET /api/quark/search?q=Go - Like()
  • GET /api/quark/range - Rango de fechas
  • GET /api/quark/in - In()

📊 Agregaciones

  • GET /api/quark/aggregations - Sum, Avg, Min, Max
  • GET /api/quark/paginated - Paginación

✏️ Modificaciones

  • POST /api/quark/update - Update()
  • DELETE /api/quark/delete/:id - Delete()

🔗 Relaciones

  • GET /api/quark/relations - Relaciones entre modelos

⚡ Transacciones

  • POST /api/quark/transaction - Transacción

🎯 Todo en Uno

  • GET /api/quark/all - Todas las características

Ejemplos de Código

Consultas Básicas

// First - Obtener primer registro
article := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    OrderBy("published_at", "DESC").
    First()

// Where encadenado
articles := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    Where("view_count", ">", 500).
    OrderBy("published_at", "DESC").
    Limit(5).
    List()

// Like - Búsqueda de texto
articles := quark.For[models.Article](ctx, client).
    Where("title", "LIKE", "%Go%").
    List()

// In - Lista de valores
articles := quark.For[models.Article](ctx, client).
    Where("category_id", "IN", []int64{1, 2, 3}).
    List()

Agregaciones

// Count
count := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    Count()

// Agregaciones manuales (Sum, Avg, Min, Max)
articles, _ := quark.For[models.Article](ctx, client).List()
totalViews := 0
for _, art := range articles {
    totalViews += art.ViewCount
}
avgViews := totalViews / len(articles)

Paginación

// Paginación con Limit
page := 1
pageSize := 5
offset := (page - 1) * pageSize

articles := quark.For[models.Article](ctx, client).
    Where("published", "=", true).
    OrderBy("published_at", "DESC").
    Limit(pageSize).
    List()

total, _ := quark.For[models.Article](ctx, client).Count()
totalPages := (int(total) + pageSize - 1) / pageSize

Modificaciones

// Create
article := models.Article{
    Title: "Nuevo Artículo",
    Slug: "nuevo-articulo",
    Published: true,
}
quark.For[models.Article](ctx, client).Create(&article)

// Update
article.ViewCount++
article.UpdatedAt = time.Now()
quark.For[models.Article](ctx, client).Update(&article)

// Delete
articles, _ := quark.For[models.Article](ctx, client).
    Where("id", "=", id).List()
quark.For[models.Article](ctx, client).Delete(&articles[0])

Relaciones

// Obtener artículo con relaciones
articles, _ := quark.For[models.Article](ctx, client).
    Where("id", "=", articleID).List()

// Get author
authors, _ := quark.For[models.Author](ctx, client).
    Where("id", "=", articles[0].AuthorID).List()

// Get category
categories, _ := quark.For[models.Category](ctx, client).
    Where("id", "=", articles[0].CategoryID).List()

// Get comments
comments, _ := quark.For[models.Comment](ctx, client).
    Where("article_id", "=", articles[0].ID).
    Where("approved", "=", true).List()
